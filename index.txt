// script.js
document.getElementById('wheel').addEventListener('click', function() {
    let degrees = Math.floor(Math.random() * 360) + 720; // Random spin + at least two full rotations
    this.style.transition = 'transform 4s ease-out';
    this.style.transform = `rotate(${degrees}deg)`;

    // Timeout to reset the wheel position quietly
    setTimeout(() => {
        this.style.transition = 'none';
        this.style.transform = 'rotate(0deg)';
    }, 4000);

    // Determine the winner based on the degrees
    let segmentIndex = Math.floor((degrees % 360) / 22.5);
    let selectedSegment = this.children[segmentIndex];
    let resultPopup = document.getElementById('result-popup');

    setTimeout(() => {
        resultPopup.textContent = selectedSegment.textContent;
        resultPopup.classList.remove('hidden');
        setTimeout(() => {
            resultPopup.classList.add('hidden');
        }, 5000);
    }, 4000);
});


</script>
